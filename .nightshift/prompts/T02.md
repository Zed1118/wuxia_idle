```dart
  test('T02 新增 5 武学领悟联结 5 encounter_skill (id 一致 + skill 存在)', () {
    final newInsights = [
      ('qing_lin_si_yu', 'skill_encounter_tian_xin_ting_yu'),
      ('gu_dao_chi_jian', 'skill_encounter_gu_dao_jian_yi'),
      ('mo_jian_ru_yu', 'skill_encounter_mo_jian_xiang_xin'),
      ('xue_ye_xing_kong', 'skill_encounter_xue_ye_xing_yi'),
      ('chuan_long_dan_xin', 'skill_encounter_chuan_long_xiao_ge'),
    ];
    for (final (encId, skillId) in newInsights) {
      final enc = encounters.firstWhere((e) => e.id == encId);
      expect(enc.type, EncounterType.techniqueInsight);
      final unlockOutcome = enc.outcomeMapping['insight_success']!;
      expect(unlockOutcome.skillId, skillId,
          reason: '$encId insight_success 应解锁 $skillId');
      expect(encounterSkills.any((s) => s.id == skillId), isTrue,
          reason: '$skillId 应在 encounter_skills.yaml 已定义');
    }
  });

  test('T02 新增 4 基础奇遇 fortuneEvent (无 festivalRequired)', () {
    final newFortunes = ['qin_lou_fang_you', 'gu_si_qiu_shu', 'ma_kuai_song_hua', 'xian_zhou_yi_lu'];
    for (final id in newFortunes) {
      final enc = encounters.firstWhere((e) => e.id == id);
      expect(enc.type, EncounterType.fortuneEvent);
      expect(enc.trigger.festivalRequired, isNull, reason: '$id 是非节日 fortuneEvent');
    }
  });
```

**关键**:测试用的变量名(`encounters` / `encounterSkills`)需要与既有 loader test 头部 import + setUpAll 一致。**先读 loader test 头部,沿用既有 fixture**,不要新建 loader。
