### B4: sect_providers Isar 真接 + sect_screen 真持久化

`lib/features/sect/application/sect_providers.dart`:

```dart
@riverpod
Stream<Sect?> currentSect(CurrentSectRef ref) {
  return IsarSetup.isar.sects
      .filter()
      .idEqualTo(1)  // Demo 单 sect 默认 id=1
      .watch(fireImmediately: true)
      .map((list) => list.isEmpty ? null : list.first);
}

@riverpod
class ResolveSectEvent extends _$ResolveSectEvent {
  @override
  AsyncValue<void> build() => const AsyncData(null);

  Future<void> resolve({
    required Sect sect,
    required SectEvent event,
    required SectOutcome outcome,
  }) async {
    final service = ref.read(sectEventServiceProvider);
    final clock = ref.read(systemClockProvider);
    final (newSect, newEvent) = service.resolve(
      sect: sect,
      event: event,
      outcome: outcome,
      now: clock.now(),
    );
    await IsarSetup.isar.writeTxn(() async {
      await IsarSetup.isar.sects.put(newSect);
      await IsarSetup.isar.sectEvents.put(newEvent);
    });
  }
}
```
