```dart
  test('hot-loop C1: synergy_gang_yin_hu_zhi 6 字段 ≤ §5.4 红线', () {
    final s = synergies.firstWhere((s) => s.id == 'synergy_gang_yin_hu_zhi');
    expect(s.multipliers.defensePct, 0.15);
    expect(s.multipliers.hpPct, 0.15);
    expect(s.multipliers.isWithinRedLine(), isTrue);
    // realm yiLiu base defense 0.20 + synergy 0.15 = 0.35 ≤ §5.5 红线 0.65
    expect(0.20 + s.multipliers.defensePct, lessThanOrEqualTo(0.65));
  });

  test('hot-loop C2: synergy_ling_gang_hui_liu attack+speed 双维 ≤ 红线', () {
    final s = synergies.firstWhere((s) => s.id == 'synergy_ling_gang_hui_liu');
    expect(s.multipliers.attackPct, 0.10);
    expect(s.multipliers.speedPct, 0.20);
    expect(s.multipliers.isWithinRedLine(), isTrue);
    // attack baseAtk 1500 × (1 + 0.10) = 1650 ≤ §5.4 装备攻击 2000 红线
  });

  test('hot-loop C3: synergy_ling_yin_gui_yi speed+ifMax 双维 ≤ 红线', () {
    final s = synergies.firstWhere((s) => s.id == 'synergy_ling_yin_gui_yi');

    expect(s.multipliers.internalForceMaxPct, 0.20);

    expect(12000 * (1 + s.multipliers.internalForceMaxPct),

  });
```
