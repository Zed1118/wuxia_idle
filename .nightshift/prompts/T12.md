## 操作

### Step 1: Sect @Collection + 2 enum(`lib/features/sect/domain/sect.dart` 新建)

完全沿 spec §2 体例。Q3.A 字段名 `sectReputation`,Q4.B 字段名 `sectLevel`,Q5.B `founderId` 弱挂。

```dart
import 'package:isar/isar.dart';

part 'sect.g.dart';

enum SectEventType { tournament, mission, crisis }
enum SectEventStatus { pending, resolved, expired }

@Collection()
class Sect {
  Id id = Isar.autoIncrement;
  late String name;
  late int founderId;                  // 弱挂 LineageMember.id 不强校验(Q5.B)
  late int sectLevel;                  // 1-7 沿七阶(Q4.B 抽象)
  late int sectReputation;             // 0-100 独立轴(Q3.A · 字段名 ≠ P1.2 reputation)
  late int totalWins;                  // 累计 tournament 胜 · 驱动 level 升级
  late DateTime createdAt;
  DateTime? lastEventAt;               // cooldown 锚 + decay 30 天起算
}
```

### Step 2: SectEvent @Collection + composite index(`lib/features/sect/domain/sect_event.dart` 新建)

```dart
import 'package:isar/isar.dart';
import 'sect.dart';                    // SectEventType / SectEventStatus enum 来源

part 'sect_event.g.dart';

@Collection()
class SectEvent {
  Id id = Isar.autoIncrement;
  @Index(composite: [CompositeIndex('triggeredAt')])
  late int sectId;                     // composite (sectId, triggeredAt) O(log n) 时序查询
  @Enumerated(EnumType.name) late SectEventType type;     // tournament(Demo)· mission/crisis 留扩
  @Enumerated(EnumType.name) late SectEventStatus status;
  late DateTime triggeredAt;
  DateTime? resolvedAt;
  late String narrativeId;             // FK data/lore/sect_event/<id>.yaml
  int? reputationDelta;
}
```

import 'package:flutter_test/flutter_test.dart';
import 'package:wuxia_idle/features/sect/domain/sect.dart';
import 'package:wuxia_idle/features/sect/domain/sect_event.dart';

void main() {
  group('P3.4 sect_event schema 红线', () {
    test('R4.1 Sect 字段冻结(spec §2)', () {
      final s = Sect()
        ..name = '无名宗'
        ..founderId = 1
        ..sectLevel = 1
        ..sectReputation = 50
        ..totalWins = 0
        ..createdAt = DateTime(2026, 5, 24);
      expect(s.sectLevel, 1);
      expect(s.sectReputation, 50);
      expect(s.lastEventAt, null);          // optional cooldown 锚
    });

    test('R4.2 SectEvent 字段 + 2 enum 冻结', () {
      final e = SectEvent()
        ..sectId = 1
        ..type = SectEventType.tournament
        ..status = SectEventStatus.pending
        ..triggeredAt = DateTime(2026, 5, 24)
        ..narrativeId = 'tournament_01';
      expect(e.type, SectEventType.tournament);
      expect(e.status, SectEventStatus.pending);
      expect(e.resolvedAt, null);
    });

    test('R4.3 SectEventType 3 枚 / SectEventStatus 3 枚', () {
      expect(SectEventType.values.length, 3);
      expect(SectEventType.values.contains(SectEventType.tournament), true);
      expect(SectEventType.values.contains(SectEventType.mission), true);
      expect(SectEventType.values.contains(SectEventType.crisis), true);
      expect(SectEventStatus.values.length, 3);
      expect(SectEventStatus.values.contains(SectEventStatus.expired), true);
    });

    // 可选 R4.4 数值红线 clamp 域(不实例化 service,只声明 spec 锚点)
    test('R4.4 sectLevel/sectReputation 数值域(spec §3 yaml 锚点)', () {
      // sectLevel ∈ [1,7] 沿七阶 / sectReputation ∈ [0,100] 独立轴
      // 真 clamp 在 Batch 2.2 SectEventService 实装,此测仅声明域(spec ack)
      const sectLevelMax = 7;
      const sectLevelMin = 1;
      const sectReputationMax = 100;
      const sectReputationMin = 0;
      expect(sectLevelMax, 7);
      expect(sectLevelMin, 1);
      expect(sectReputationMax, 100);
      expect(sectReputationMin, 0);
    });
  });
}
