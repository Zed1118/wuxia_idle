### Step 19: R4 loader 测(2 文件)

`test/features/pvp/pvp_narrative_loader_test.dart`:

```dart
import 'dart:io';

import 'package:flutter_test/flutter_test.dart';
import 'package:yaml/yaml.dart';

void main() {
  group('R4 PVP narrative loader', () {
    final pvpDir = Directory('data/lore/pvp');
    final allYamls = pvpDir.listSync().whereType<File>().toList();

    test('PVP narrative 文件数 ≥ 11(初战 + 连胜 2 + 升级 6 + 降段 + 月榜)', () {
      expect(allYamls.length, greaterThanOrEqualTo(11));
    });

    for (final f in allYamls) {
      final basename = f.uri.pathSegments.last.replaceAll('.yaml', '');
      test('$basename schema 合法(id 等 文件名 + trigger.kind 非空 + title + opening 非空 + opening ≤ 6 行)', () {
        final doc = loadYaml(f.readAsStringSync()) as Map;
        expect(doc['id'], equals(basename));
        expect(doc['trigger']?['kind'], isNotNull);
        expect((doc['title'] as String).isNotEmpty, isTrue);
        final opening = doc['opening'] as String;
        expect(opening.isNotEmpty, isTrue);
        expect(opening.split('\n').where((l) => l.trim().isNotEmpty).length, lessThanOrEqualTo(6));
      });
    }

    test('黑名单词 0(霸气/逆天/史诗/legendary)', () {
      const blacklist = ['霸气', '逆天', '史诗', '神级', '无敌', '最强', '究极', 'legendary', 'epic', '刀光剑影', '血雨腥风'];
      for (final f in allYamls) {
        final content = f.readAsStringSync();
        for (final w in blacklist) {
          expect(content, isNot(contains(w)), reason: '${f.path} 含黑名单词 $w');
        }
      }
    });
  });
}
```
