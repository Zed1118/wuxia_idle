```dart
/// PVP 异步快照对决策略(1.0 P3.3 §12.3)。
///
/// **组合委派架构**(沿 LightFootStrategy 体例 / memory `feedback_avoid_over_engineer_abstraction`):
/// 内部持 `const DefaultGroundStrategy _delegate`,zero 代码重复;tick/
/// requestUltimate 直接委派,runToEnd 入口把 opponentTeam 装到 rightTeam,
/// 再委派 _delegate.runToEnd。
///
/// **数值红线**:0 引入 `attackPowerMultiplier`(§5.4 反 ELO 段位 buff 越权),
/// ELO 仅 UI 段位词,不进 BattleEngine。
///
/// **immutable**:opponentTeam 在 ctor 注入,runToEnd 入口 idempotent。
class PvpStrategy implements BattleStrategy {
  /// 对手阵容(NoopPvpSync 本地 mirror 生成 / Phase 5 SupabasePvpSync 从快照解码)。
  final List<BattleCharacter> opponentTeam;

  static const _delegate = DefaultGroundStrategy();

  const PvpStrategy({required this.opponentTeam});

  @override
  BattleState runToEnd(
    BattleState initial,
    NumbersConfig n, {
    int maxTicks = 1000,
    Random? rng,
  }) {
    // 注入 opponentTeam 到 rightTeam(rest 沿 initial 即可)
    final hydrated = initial.copyWith(rightTeam: opponentTeam);
    return _delegate.runToEnd(hydrated, n, maxTicks: maxTicks, rng: rng);
  }

  @override
  BattleState tick(BattleState state, NumbersConfig n, {Random? rng}) =>
      _delegate.tick(state, n, rng: rng);

  @override
  BattleState requestUltimate(BattleState state, int characterId) =>
      _delegate.requestUltimate(state, characterId);
}
```

> **注**:确认 `BattleState.copyWith(rightTeam: ...)` 真存在(Phase 0 grep);若不存在改用 `BattleState(...)` 全字段重建。BattleStrategy 接口三 method 签名以 `battle_strategy.dart` 实际为准(尤其 `requestUltimate` 签名)。

### Step 3: PvpService + ELO calc(`lib/features/pvp/application/pvp_service.dart` + `pvp_elo.dart` 新建)

**ELO 算法**纯函数抽出 `pvp_elo.dart`(便于 R3 测族单测):

```dart
// lib/features/pvp/application/pvp_elo.dart
double expectedScore(int selfElo, int oppElo) =>
    1.0 / (1.0 + pow(10, (oppElo - selfElo) / 400.0));

/// actualScore: 1.0 = win / 0.5 = draw / 0.0 = loss
int eloDelta({
  required int selfElo,
  required int oppElo,
  required double actualScore,
  required int kFactor,        // 默认 32(numbers.yaml pvp.elo.k_factor)
}) {
  final expected = expectedScore(selfElo, oppElo);
  return (kFactor * (actualScore - expected)).round();
}
```

**PvpService.match**:

```dart
// lib/features/pvp/application/pvp_service.dart
class PvpService {
  final PvpSyncService sync;
  final NumbersConfig numbers;
  // 可注入 Isar / repo / 其他 dep(Phase 4 wire 进 Riverpod)

  PvpService({required this.sync, required this.numbers});

  /// 单场异步对决主入口。
  /// 返 PvpRecord(已计算 ELO delta · 调 sync.uploadResult)。
  Future<PvpRecord> match({
    required int playerId,
    required int playerElo,
    required List<BattleCharacter> playerTeam,
  }) async {
    // 1. find opponent
    final eloWindow = numbers.pvp.matchRange.eloWindow;
    final oppTeam = await sync.findOpponent(playerElo: playerElo, eloWindow: eloWindow);

    // 2. run battle
    final strat = PvpStrategy(opponentTeam: oppTeam);
    final initial = _buildInitialState(playerTeam, oppTeam);
    final result = strat.runToEnd(initial, numbers).result;

    // 3. ELO calc
    final actualScore = _scoreFromResult(result);
    final oppElo = playerElo;  // mirror Noop 阶段对手同 ELO · Phase 5 接真 elo
    final delta = eloDelta(
      selfElo: playerElo,
      oppElo: oppElo,
      actualScore: actualScore,
      kFactor: numbers.pvp.elo.kFactor,
    );

    // 4. 装 PvpRecord
    final record = PvpRecord()
      ..matchId = _uuid()
      ..playerId = playerId
      ..opponentSnapshotId = 0     // Phase 3 Noop 未真持久化 snapshot
      ..leftSnapshotId = 0
      ..winnerId = _winnerIdFromResult(result)
      ..playerEloBefore = playerElo
      ..playerEloAfter = playerElo + delta
      ..eloDelta = delta
      ..timestamp = DateTime.now();

    // 5. upload(Noop = 0 网)
    await sync.uploadResult(record);
    return record;
  }
}
```
